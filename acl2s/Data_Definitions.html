8 Data Definitions

syntax

(defdata name type)

 
type = singleton
  | simple-type
  | (enum lst-expr)
  | (oneof type type)
  | (range numeric-type range-constraints)
  | (list type ...)
  | (cons type type)
  | (listof type)
  | (record (field-name . type) ...)
     
simple-type = all
  | boolean
  | symbol
  | string
  | tl
  | numeric-type
     
numeric-type = rational
  | integer
  | nat
  | pos
     
range-constraints = (num-expr lt _ lt num-expr)
  | (num-expr lt _)
  | (_ lt num-expr)
     
lt = <
  | <=
Defines the function namep as a recognizer for the given type.

Singletons:
> (defdata one 1)
> (onep 1)

t

List Types:
> (defdata natlist (listof nat))
> (natlistp nil)

t

> (natlistp (list 2 3 5 8))

t

> (natlistp (list 2 3 -5 8))

nil
